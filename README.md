function dnaStrand(dna) {
    // Creating an object
    const complements = {
        'A': 'T',
        'T': 'A',
        'C': 'G',
        'G': 'C'
    };
    // Use map to replace each character by a key and combine the result back into a string
    return dna.split('').map(nucl => complements[nucl]).join('');
}
